## 6. 自检画框脚本(核对光标必用)

```bash
./.venv/bin/python - <<'EOF'
import cv2, json, sqlite3
VID, T = "vid_605df2fff231", 11.0   # 换时间点
db = sqlite3.connect("storage/dreamhome.db")
rows = db.execute("SELECT category, frames_json FROM tracks WHERE video_id=?", (VID,)).fetchall()
cap = cv2.VideoCapture("storage/downloads/03d305377517.mp4")
cap.set(cv2.CAP_PROP_POS_MSEC, T*1000); ok, img = cap.read(); H, W = img.shape[:2]
for cat, fj in rows:
    near = [p for p in json.loads(fj) if abs(p["t"]-T) <= 0.21]
    if not near: continue
    x,y,w,h = min(near, key=lambda p: abs(p["t"]-T))["bbox"]
    cv2.rectangle(img,(int(x*W),int(y*H)),(int((x+w)*W),int((y+h)*H)),(0,255,0),2)
    cv2.putText(img,cat,(int(x*W),int(y*H)-4),cv2.FONT_HERSHEY_SIMPLEX,0.7,(0,0,255),2)
cv2.imwrite(f"/tmp/audit_{T}.jpg", img); print("saved")
EOF
```
